{{/* Generate the description meta tag. */}} {{- if or .Summary .Description -}} {{- $summary := "" -}} {{- if .IsPage -}} {{- if .Params.summary -}} {{- $summary = .Params.summary | plainify | htmlUnescape -}} {{- $summary = replaceRE "[\r\n]+" " " $summary -}} {{- else -}} {{- $summaryLength := int (default 70 .Site.Params.summary.length) -}} {{- $summary = .Content | plainify | htmlUnescape -}} {{- $summary = $summary | truncate $summaryLength -}} {{- $summary = replaceRE "[\r\n]+" " " $summary -}} {{- end -}} {{- else -}} {{- $summary = .Summary -}} {{- end -}} {{- $summary = trim ($summary | htmlUnescape) " " -}} {{- end -}} {{/* Generate the noindex meta tag. */}} {{- if .Params.noindex -}} {{- end -}}
